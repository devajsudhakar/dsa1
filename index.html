const { MongoClient } = require("mongodb");
const readlineSync = require("readline-sync");
const url = "mongodb://localhost:27017";
const client = new MongoClient(url);
async function main() {
const partialName = readlineSync.question("Enter Partial Name: ");
try {
await client.connect();
const collection = client.db('school').collection('students');
const query = { name: { $regex: partialName, $options: 'i' } };
const students = await collection.find(query).toArray();
console.log(`The Students whose partial name is ${partialName} are: \n`, students);
} catch (error) {
console.log("Something went wrong while searching\n\n", error);
} finally {
await client.close();
}
}
main();

const path = require("path");
const fs = require("fs");
const rls = require("readline-sync");
const dataPath = "./data";
if (!fs.existsSync(dataPath))
fs.mkdirSync(dataPath);
function createNewFile() {
const fileName = rls.question("Enter File name: ");
const content = rls.question("Enter File Content: ");
const filePath = path.join(dataPath, fileName);
try {
fs.writeFileSync(filePath, content);
console.log("File Succesfully Created");
} catch (error) {
console.log("Something went wrong while creating new file: \n", error.message);
}
}
function readFile() {
const fileName = rls.question("Enter file name to read: ");
const filePath = path.join(dataPath, fileName);
try {
const data = fs.readFileSync(filePath, 'utf-8');
console.log(data);
} catch (error) {
console.log("Something went wrong while reading file\n", error.message);
}
}
function updateFile() {
const fileName = rls.question("Enter file name to update: ");
const content = rls.question("Enter the file content: ");
const filePath = path.join(dataPath, fileName);
try {
fs.writeFileSync(filePath, content);
console.log("File upserted Successfully");
} catch (error) {
console.log("Something went wrong while updating file\n", error.message);
}
}
function deleteFile() {
const fileName = rls.question("Enter file name to delete: ");
const filePath = path.join(dataPath, fileName);
try {
fs.unlinkSync(filePath);
console.log("File deleted Successfully");
} catch (error) {
console.log("Something went wrong while deleting file\n", error.message);
}
}
(function main() {
const choice = rls.question("Enter your choice: (CRUD)");
switch (choice) {
case '1':
createNewFile();
break;
case '2':
readFile();
break;
case '3':
updateFile();
break;
case '4':
deleteFile();
break;
default:
console.log("Invalid Choice!!\n");
}
})();

const express = require("express");
const bodyparser = require("body-parser");
const app = express();
app.use(bodyparser.urlencoded())
app.use(express.static('public'));
let users = [];
app.get("/login", (req, res) => {
res.sendFile(__dirname + "/public/login.html");
})
app.get("/signup", (req, res) => {
res.sendFile(__dirname + "/public/signup.html")
})
app.get("/home", (req, res) => {
res.sendFile(__dirname + "/public/home.html");
})
app.post("/login", (req, res) => {
const { email, password } = req.body;
if (email === undefined || password === undefined)
res.send("Enter valid emaial and password");
if (users.findIndex((u) => u.email === email && u.password === password) != -1)
res.redirect("/home");
else
res.send("Invalid email-id or password");
})
app.post("/signup", (req, res) => {
const { email, password } = req.body;
if (email && password)
users.push({ email, password });
else res.send("Enter valid emaial and password");
res.redirect("/login");
})
app.get("/allusers", (req, res) => {
res.send(users);
})
app.listen(3000, () => {
console.log("Server Up and Running on Port 3000");
})

const ans = ["<h3>2</h3>"];
for (let i = 3; i <= 100; i++) {
let isPrime = true;
for (let j = 2; j < i / 2; j++)
if (i % j === 0) {
isPrime = false;
break;
}
if (isPrime) ans.push(`<h3>${i}</h3>`);
}
const html = "<h1>All Primes from 1 to 100 <h1>" + ans.join("");
res.send(html);
})

import { useState } from "react";
export default function useFilterApp() {
const [newItem, setNewItem] = useState("");
const [searchItem, setSearchItem] = useState("");
const [items, setItems] = useState(["Apple", "Bannana", "Mango"]);
const filteredItems = items.filter(item =>
item.toLowerCase().includes(searchItem.toLowerCase())
);
return (
<>
<label htmlFor="search">Search for: </label>
<input
type="text"
name="search"
id="search"
value={searchItem} onChange={(e) => {
setSearchItem(e.target.value);
}}
/>
<hr />
<h2>Filtered Item</h2>
{
filteredItems.length === 0 ? (
<h3>No Items Found</h3>
) :
(filteredItems.map((item, idx) => <li key={idx}>{item}</li>))
}
<hr />
<label htmlFor="new">Add Item</label>
<input
type="text"
name="new"
id="new"
value={newItem}
onChange={(e) => {
setNewItem(e.target.value);
}}
/>
<button onClick={() => { setItems(items => [...items, newItem]);
setNewItem("");
}}>Add</button>
</>
)
}
